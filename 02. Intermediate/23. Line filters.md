# Line filters

- Read contents of a file line by line and then process each line to get a more meaningful output like removing empty lines or filter lines with some keywords.

```go
func main() {
	file, err := os.Open("logs.txt")

	if err != nil {
		fmt.Println("Error while opening the file")
		return
	}

	defer file.Close()

	scanner := bufio.NewScanner(file)

	keyword := "Lorem"

	for scanner.Scan() {
		line := scanner.Text()
		if strings.Contains(line, keyword) {
			fmt.Println("=====>", line)
		}
	}

	err = scanner.Err()

	if err != nil {
		fmt.Println("Error while reading file content")
		return
	}

}
```
