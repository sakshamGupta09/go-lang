# Channel Synchronization


## Wait for a task to complete

```go
func main() {
	done := make(chan bool)

	go func() {
		fmt.Println("WORK STARTED ....")
		time.Sleep(2 * time.Second)
		done <- true
	}()

	<-done
	fmt.Println("TASK IS NOW COMPLETE")
}
```

```go
func main() {
	goroutinesCount := 3
	done := make(chan int, goroutinesCount)

	for i := range goroutinesCount {
		go func(id int) {
			fmt.Println("STARTING Goroutine number", id)
			time.Sleep(1 * time.Second)
			done <- id
		}(i)
	}

	for range goroutinesCount {
		fmt.Println(<-done)
	}

}
```
